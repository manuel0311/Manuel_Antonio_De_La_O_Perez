  54       * Number of rows in the result set
  55       *
  56       * @return    int
  57       */
  58  	public function num_rows()
  59      {
  60          if (is_int($this->num_rows))
  61          {
  62              return $this->num_rows;
  63          }
  64          elseif (($this->num_rows = odbc_num_rows($this->result_id)) !== -1)
  65          {
  66              return $this->num_rows;
  67          }
  68  
  69          // Work-around for ODBC subdrivers that don't support num_rows()
  70          if (count($this->result_array) > 0)
  71          {
  72              return $this->num_rows = count($this->result_array);
  73          }
  74          elseif (count($this->result_object) > 0)
  75          {
  76              return $this->num_rows = count($this->result_object);
  77          }
  78  
  79          return $this->num_rows = count($this->result_array());
  80      }

 108          if ($num_fields > 0)
 109          {
 110              for ($i = 1; $i <= $num_fields; $i++)
 111              {
 112                  $field_names[] = odbc_field_name($this->result_id, $i);
 113              }

 122       * Field data
 123       *
 124       * Generates an array of objects containing field meta-data
 125       *
 126       * @return    array
 127       */
 128  	public function field_data()
 129      {
 130          $retval = array();
 131          for ($i = 0, $odbc_index = 1, $c = $this->num_fields(); $i < $c; $i++, $odbc_index++)
 132          {
 133              $retval[$i]            = new stdClass();
 134              $retval[$i]->name        = odbc_field_name($this->result_id, $odbc_index);
 135              $retval[$i]->type        = odbc_field_type($this->result_id, $odbc_index);
 136              $retval[$i]->max_length        = odbc_field_len($this->result_id, $odbc_index);
 137              $retval[$i]->primary_key    = 0;
 138              $retval[$i]->default        = '';
 139          }
 140  
 141          return $retval;
 142      }
