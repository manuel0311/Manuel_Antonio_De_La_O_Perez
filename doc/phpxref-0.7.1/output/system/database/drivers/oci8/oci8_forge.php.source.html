  95       *
  96       * @param    string    $alter_type    ALTER type
  97       * @param    string    $table        Table name
  98       * @param    mixed    $field        Column definition
  99       * @return    string|string[]
 100       */
 101  	protected function _alter_table($alter_type, $table, $field)
 102      {
 103          if ($alter_type === 'DROP')
 104          {
 105              return parent::_alter_table($alter_type, $table, $field);
 106          }
 107          elseif ($alter_type === 'CHANGE')
 108          {
 109              $alter_type = 'MODIFY';
 110          }
 111  
 112          $sql = 'ALTER TABLE '.$this->db->escape_identifiers($table);
 113          $sqls = array();
 114          for ($i = 0, $c = count($field); $i < $c; $i++)
 115          {
 116              if ($field[$i]['_literal'] !== FALSE)
 117              {
 118                  $field[$i] = "\n\t".$field[$i]['_literal'];
 119              }
 120              else
 121              {
 122                  $field[$i]['_literal'] = "\n\t".$this->_process_column($field[$i]);
 123  
 124                  if ( ! empty($field[$i]['comment']))
 125                  {
 126                      $sqls[] = 'COMMENT ON COLUMN '
 127                          .$this->db->escape_identifiers($table).'.'.$this->db->escape_identifiers($field[$i]['name'])
 128                          .' IS '.$field[$i]['comment'];
 129                  }
 130  
 131                  if ($alter_type === 'MODIFY' && ! empty($field[$i]['new_name']))
 132                  {
 133                      $sqls[] = $sql.' RENAME COLUMN '.$this->db->escape_identifiers($field[$i]['name'])
 134                          .' TO '.$this->db->escape_identifiers($field[$i]['new_name']);
 135                  }
 136  
 137                  $field[$i] = "\n\t".$field[$i]['_literal'];
 138              }
 139          }
 140  
 141          $sql .= ' '.$alter_type.' ';
 142          $sql .= (count($field) === 1)
 143                  ? $field[0]
 144                  : '('.implode(',', $field).')';
 145  
 146          // RENAME COLUMN must be executed after MODIFY
 147          array_unshift($sqls, $sql);
 148          return $sqls;
 149      }
