 105  	protected function _alter_table($alter_type, $table, $field)
 106      {
 107          if (in_array($alter_type, array('DROP', 'ADD'), TRUE))
 108          {
 109              return parent::_alter_table($alter_type, $table, $field);
 110          }
 111  
 112          $sql = 'ALTER TABLE '.$this->db->escape_identifiers($table);
 113          $sqls = array();
 114          for ($i = 0, $c = count($field); $i < $c; $i++)
 115          {
 116              if ($field[$i]['_literal'] !== FALSE)
 117              {
 118                  $sqls[] = $sql.' CHANGE '.$field[$i]['_literal'];
 119              }
 120              else
 121              {
 122                  $alter_type = empty($field[$i]['new_name']) ? ' MODIFY ' : ' CHANGE ';
 123                  $sqls[] = $sql.$alter_type.$this->_process_column($field[$i]);
 124              }
 125          }
 126  
 127          return $sqls;
