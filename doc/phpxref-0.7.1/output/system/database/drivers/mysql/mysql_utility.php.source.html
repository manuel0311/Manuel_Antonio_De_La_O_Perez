  72      /**
  73       * Export
  74       *
  75       * @param    array    $params    Preferences
  76       * @return    mixed
  77       */
  78  	protected function _backup($params = array())
  79      {
  80          if (count($params) === 0)
  81          {
  82              return FALSE;
  83          }
  84  
  85          // Extract the prefs for simplicity
  86          extract($params);
  87  
  88          // Build the output
  89          $output = '';
  90  
  91          // Do we need to include a statement to disable foreign key checks?
  92          if ($foreign_key_checks === FALSE)
  93          {
  94              $output .= 'SET foreign_key_checks = 0;'.$newline;
  95          }
  96  
  97          foreach ( (array) $tables as $table)
  98          {
  99              // Is the table in the "ignore" list?
 100              if (in_array($table, (array) $ignore, TRUE))
 101              {
 102                  continue;
 103              }
 104  
 105              // Get the table schema
 106              $query = $this->db->query('SHOW CREATE TABLE '.$this->db->escape_identifiers($this->db->database.'.'.$table));
 107  
 108              // No result means the table name was invalid
 109              if ($query === FALSE)
 110              {
 111                  continue;
 112              }
 113  
 114              // Write out the table schema
 115              $output .= '#'.$newline.'# TABLE STRUCTURE FOR: '.$table.$newline.'#'.$newline.$newline;
 116  
 117              if ($add_drop === TRUE)
 118              {
 119                  $output .= 'DROP TABLE IF EXISTS '.$this->db->protect_identifiers($table).';'.$newline.$newline;
 120              }
 121  
 122              $i = 0;
 123              $result = $query->result_array();
 124              foreach ($result[0] as $val)
 125              {
 126                  if ($i++ % 2)
 127                  {
 128                      $output .= $val.';'.$newline.$newline;
 129                  }
 130              }
 131  
 132              // If inserts are not needed we're done...
 133              if ($add_insert === FALSE)
 134              {
 135                  continue;
 136              }
 137  
 138              // Grab all the data from the current table
 139              $query = $this->db->query('SELECT * FROM '.$this->db->protect_identifiers($table));
 140  
 141              if ($query->num_rows() === 0)
 142              {
 143                  continue;
 144              }
 145  
 146              // Fetch the field names and determine if the field is an
 147              // integer type. We use this info to decide whether to
 148              // surround the data with quotes or not
 149  
 150              $i = 0;
 151              $field_str = '';
 152              $is_int = array();
 153              while ($field = mysql_fetch_field($query->result_id))
 154              {
 155                  // Most versions of MySQL store timestamp as a string
 156                  $is_int[$i] = in_array(strtolower(mysql_field_type($query->result_id, $i)),
 157                              array('tinyint', 'smallint', 'mediumint', 'int', 'bigint'), //, 'timestamp'),
 158                              TRUE);
 159  
 160                  // Create a string of field names
 161                  $field_str .= $this->db->escape_identifiers($field->name).', ';
 162                  $i++;
 163              }
 164  
 165              // Trim off the end comma
 166              $field_str = preg_replace('/, $/' , '', $field_str);
 167  
 168              // Build the insert string
 169              foreach ($query->result_array() as $row)
 170              {
 171                  $val_str = '';
 172  
 173                  $i = 0;
 174                  foreach ($row as $v)
 175                  {
 176                      // Is the value NULL?
 177                      if ($v === NULL)
 178                      {
 179                          $val_str .= 'NULL';
 180                      }
 181                      else
 182                      {
 183                          // Escape the data if it's not an integer
 184                          $val_str .= ($is_int[$i] === FALSE) ? $this->db->escape($v) : $v;
 185                      }
 186  
 187                      // Append a comma
 188                      $val_str .= ', ';
 189                      $i++;
 190                  }
 191  
 192                  // Remove the comma at the end of the string
 193                  $val_str = preg_replace('/, $/' , '', $val_str);
 194  
 195                  // Build the INSERT string
 196                  $output .= 'INSERT INTO '.$this->db->protect_identifiers($table).' ('.$field_str.') VALUES ('.$val_str.');'.$newline;
 197              }
 198  
 199              $output .= $newline.$newline;
 200          }
 201  
 202          // Do we need to include a statement to re-enable foreign key checks?
 203          if ($foreign_key_checks === FALSE)
 204          {
 205              $output .= 'SET foreign_key_checks = 1;'.$newline;
 206          }
 207  
 208          return $output;
 209      }
