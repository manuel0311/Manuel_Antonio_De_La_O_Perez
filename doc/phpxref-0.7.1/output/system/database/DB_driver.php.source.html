 380      /**
 381       * Initialize Database Settings
 382       *
 383       * @return    bool
 384       */
 385  	public function initialize()
 386      {
 387          /* If an established connection is available, then there's
 388           * no need to connect and select the database.
 389           *
 390           * Depending on the database driver, conn_id can be either
 391           * boolean TRUE, a resource or an object.
 392           */
 393          if ($this->conn_id)
 394          {
 395              return TRUE;
 396          }
 397  
 398          // ----------------------------------------------------------------
 399  
 400          // Connect to the database and set the connection ID
 401          $this->conn_id = $this->db_connect($this->pconnect);
 402  
 403          // No connection resource? Check if there is a failover else throw an error
 404          if ( ! $this->conn_id)
 405          {
 406              // Check if there is a failover set
 407              if ( ! empty($this->failover) && is_array($this->failover))
 408              {
 409                  // Go over all the failovers
 410                  foreach ($this->failover as $failover)
 411                  {
 412                      // Replace the current settings with those of the failover
 413                      foreach ($failover as $key => $val)
 414                      {
 415                          $this->$key = $val;
 416                      }
 417  
 418                      // Try to connect
 419                      $this->conn_id = $this->db_connect($this->pconnect);
 420  
 421                      // If a connection is made break the foreach loop
 422                      if ($this->conn_id)
 423                      {
 424                          break;
 425                      }
 426                  }
 427              }
 428  
 429              // We still don't have a connection?
 430              if ( ! $this->conn_id)
 431              {
 432                  log_message('error', 'Unable to connect to the database');
 433  
 434                  if ($this->db_debug)
 435                  {
 436                      $this->display_error('db_unable_to_connect');
 437                  }
 438  
 439                  return FALSE;
 440              }
 441          }
 442  
 443          // Now we set the character set and that's all
 444          return $this->db_set_charset($this->char_set);
 445      }

 594      /**
 595       * Execute the query
 596       *
 597       * Accepts an SQL string as input and returns a result object upon
 598       * successful execution of a "read" type query. Returns boolean TRUE
 599       * upon successful execution of a "write" type query. Returns boolean
 600       * FALSE upon failure, and if the $db_debug variable is set to TRUE
 601       * will raise an error.
 602       *
 603       * @param    string    $sql
 604       * @param    array    $binds = FALSE        An array of binding data
 605       * @param    bool    $return_object = NULL
 606       * @return    mixed
 607       */
 608  	public function query($sql, $binds = FALSE, $return_object = NULL)
 609      {
 610          if ($sql === '')
 611          {
 612              log_message('error', 'Invalid query: '.$sql);
 613              return ($this->db_debug) ? $this->display_error('db_invalid_query') : FALSE;
 614          }
 615          elseif ( ! is_bool($return_object))
 616          {
 617              $return_object = ! $this->is_write_type($sql);
 618          }
 619  
 620          // Verify table prefix and replace if necessary
 621          if ($this->dbprefix !== '' && $this->swap_pre !== '' && $this->dbprefix !== $this->swap_pre)
 622          {
 623              $sql = preg_replace('/(\W)'.$this->swap_pre.'(\S+?)/', '\\1'.$this->dbprefix.'\\2', $sql);
 624          }
 625  
 626          // Compile binds if needed
 627          if ($binds !== FALSE)
 628          {
 629              $sql = $this->compile_binds($sql, $binds);
 630          }
 631  
 632          // Is query caching enabled? If the query is a "read type"
 633          // we will load the caching class and return the previously
 634          // cached query if it exists
 635          if ($this->cache_on === TRUE && $return_object === TRUE && $this->_cache_init())
 636          {
 637              $this->load_rdriver();
 638              if (FALSE !== ($cache = $this->CACHE->read($sql)))
 639              {
 640                  return $cache;
 641              }
 642          }
 643  
 644          // Save the query for debugging
 645          if ($this->save_queries === TRUE)
 646          {
 647              $this->queries[] = $sql;
 648          }
 649  
 650          // Start the Query Timer
 651          $time_start = microtime(TRUE);
 652  
 653          // Run the Query
 654          if (FALSE === ($this->result_id = $this->simple_query($sql)))
 655          {
 656              if ($this->save_queries === TRUE)
 657              {
 658                  $this->query_times[] = 0;
 659              }
 660  
 661              // This will trigger a rollback if transactions are being used
 662              if ($this->_trans_depth !== 0)
 663              {
 664                  $this->_trans_status = FALSE;
 665              }
 666  
 667              // Grab the error now, as we might run some additional queries before displaying the error
 668              $error = $this->error();
 669  
 670              // Log errors
 671              log_message('error', 'Query error: '.$error['message'].' - Invalid query: '.$sql);
 672  
 673              if ($this->db_debug)
 674              {
 675                  // We call this function in order to roll-back queries
 676                  // if transactions are enabled. If we don't call this here
 677                  // the error message will trigger an exit, causing the
 678                  // transactions to remain in limbo.
 679                  while ($this->_trans_depth !== 0)
 680                  {
 681                      $trans_depth = $this->_trans_depth;
 682                      $this->trans_complete();
 683                      if ($trans_depth === $this->_trans_depth)
 684                      {
 685                          log_message('error', 'Database: Failure during an automated transaction commit/rollback!');
 686                          break;
 687                      }
 688                  }
 689  
 690                  // Display errors
 691                  return $this->display_error(array('Error Number: '.$error['code'], $error['message'], $sql));
 692              }
 693  
 694              return FALSE;
 695          }
 696  
 697          // Stop and aggregate the query time results
 698          $time_end = microtime(TRUE);
 699          $this->benchmark += $time_end - $time_start;
 700  
 701          if ($this->save_queries === TRUE)
 702          {
 703              $this->query_times[] = $time_end - $time_start;
 704          }
 705  
 706          // Increment the query counter
 707          $this->query_count++;
 708  
 709          // Will we have a result object instantiated? If not - we'll simply return TRUE
 710          if ($return_object !== TRUE)
 711          {
 712              // If caching is enabled we'll auto-cleanup any existing files related to this particular URI
 713              if ($this->cache_on === TRUE && $this->cache_autodel === TRUE && $this->_cache_init())
 714              {
 715                  $this->CACHE->delete();
 716              }
 717  
 718              return TRUE;
 719          }
 720  
 721          // Load and instantiate the result driver
 722          $driver        = $this->load_rdriver();
 723          $RES        = new $driver($this);
 724  
 725          // Is query caching enabled? If so, we'll serialize the
 726          // result object and save it to a cache file.
 727          if ($this->cache_on === TRUE && $this->_cache_init())
 728          {
 729              // We'll create a new instance of the result object
 730              // only without the platform specific driver since
 731              // we can't use it with cached data (the query result
 732              // resource ID won't be any good once we've cached the
 733              // result object, so we'll have to compile the data
 734              // and save it)
 735              $CR = new CI_DB_result($this);
 736              $CR->result_object    = $RES->result_object();
 737              $CR->result_array    = $RES->result_array();
 738              $CR->num_rows        = $RES->num_rows();
 739  
 740              // Reset these since cached objects can not utilize resource IDs.
 741              $CR->conn_id        = NULL;
 742              $CR->result_id        = NULL;
 743  
 744              $this->CACHE->write($sql, $CR);
 745          }
 746  
 747          return $RES;
 748      }

 847      /**
 848       * Complete Transaction
 849       *
 850       * @return    bool
 851       */
 852  	public function trans_complete()
 853      {
 854          if ( ! $this->trans_enabled)
 855          {
 856              return FALSE;
 857          }
 858  
 859          // The query() function will set this flag to FALSE in the event that a query failed
 860          if ($this->_trans_status === FALSE OR $this->_trans_failure === TRUE)
 861          {
 862              $this->trans_rollback();
 863  
 864              // If we are NOT running in strict mode, we will reset
 865              // the _trans_status flag so that subsequent groups of
 866              // transactions will be permitted.
 867              if ($this->trans_strict === FALSE)
 868              {
 869                  $this->_trans_status = TRUE;
 870              }
 871  
 872              log_message('debug', 'DB Transaction Failure');
 873              return FALSE;
 874          }
 875  
 876          return $this->trans_commit();
 877      }

 893      /**
 894       * Begin Transaction
 895       *
 896       * @param    bool    $test_mode
 897       * @return    bool
 898       */
 899  	public function trans_begin($test_mode = FALSE)
 900      {
 901          if ( ! $this->trans_enabled)
 902          {
 903              return FALSE;
 904          }
 905          // When transactions are nested we only begin/commit/rollback the outermost ones
 906          elseif ($this->_trans_depth > 0)
 907          {
 908              $this->_trans_depth++;
 909              return TRUE;
 910          }
 911  
 912          // Reset the transaction failure flag.
 913          // If the $test_mode flag is set to TRUE transactions will be rolled back
 914          // even if the queries produce a successful result.
 915          $this->_trans_failure = ($test_mode === TRUE);
 916  
 917          if ($this->_trans_begin())
 918          {
 919              $this->_trans_status = TRUE;
 920              $this->_trans_depth++;
 921              return TRUE;
 922          }
 923  
 924          return FALSE;
 925      }

 931       *
 932       * @return    bool
 933       */
 934  	public function trans_commit()
 935      {
 936          if ( ! $this->trans_enabled OR $this->_trans_depth === 0)
 937          {
 938              return FALSE;

 941          elseif ($this->_trans_depth > 1 OR $this->_trans_commit())

 944              return TRUE;

 975      /**
 976       * Compile Bindings
 977       *
 978       * @param    string    the sql statement
 979       * @param    array    an array of bind data
 980       * @return    string
 981       */
 982  	public function compile_binds($sql, $binds)
 983      {
 984          if (empty($this->bind_marker) OR strpos($sql, $this->bind_marker) === FALSE)
 985          {
 986              return $sql;
 987          }
 988          elseif ( ! is_array($binds))
 989          {
 990              $binds = array($binds);
 991              $bind_count = 1;
 992          }
 993          else
 994          {
 995              // Make sure we're using numeric keys
 996              $binds = array_values($binds);
 997              $bind_count = count($binds);
 998          }
 999  
1000          // We'll need the marker length later
1001          $ml = strlen($this->bind_marker);
1002  
1003          // Make sure not to replace a chunk inside a string that happens to match the bind marker
1004          if ($c = preg_match_all("/'[^']*'|\"[^\"]*\"/i", $sql, $matches))
1005          {
1006              $c = preg_match_all('/'.preg_quote($this->bind_marker, '/').'/i',
1007                  str_replace($matches[0],
1008                      str_replace($this->bind_marker, str_repeat(' ', $ml), $matches[0]),
1009                      $sql, $c),
1010                  $matches, PREG_OFFSET_CAPTURE);
1011  
1012              // Bind values' count must match the count of markers in the query
1013              if ($bind_count !== $c)
1014              {
1015                  return $sql;
1016              }
1017          }
1018          elseif (($c = preg_match_all('/'.preg_quote($this->bind_marker, '/').'/i', $sql, $matches, PREG_OFFSET_CAPTURE)) !== $bind_count)
1019          {
1020              return $sql;
1021          }
1022  
1023          do
1024          {
1025              $c--;
1026              $escaped_value = $this->escape($binds[$c]);
1027              if (is_array($escaped_value))
1028              {
1029                  $escaped_value = '('.implode(',', $escaped_value).')';
1030              }
1031              $sql = substr_replace($sql, $escaped_value, $matches[0][$c][1], $ml);
1032          }
1033          while ($c !== 0);
1034  
1035          return $sql;
1036      }

1094       * Sets boolean and null types
1095       *
1096       * @param    string
1097       * @return    mixed
1098       */

1102          {

1104              return $str;

1375      /**
1376       * Escape the SQL Identifiers
1377       *
1378       * This function escapes column and table names
1379       *
1380       * @param    mixed
1381       * @return    mixed
1382       */
1383  	public function escape_identifiers($item)
1384      {
1385          if ($this->_escape_char === '' OR empty($item) OR in_array($item, $this->_reserved_identifiers))
1386          {
1387              return $item;
1388          }
1389          elseif (is_array($item))
1390          {
1391              foreach ($item as $key => $value)
1392              {
1393                  $item[$key] = $this->escape_identifiers($value);
1394              }
1395  
1396              return $item;
1397          }
1398          // Avoid breaking functions and literal values inside queries
1399          elseif (ctype_digit($item) OR $item[0] === "'" OR ($this->_escape_char !== '"' && $item[0] === '"') OR strpos($item, '(') !== FALSE)
1400          {
1401              return $item;
1402          }
1403  
1404          static $preg_ec = array();
1405  
1406          if (empty($preg_ec))
1407          {
1408              if (is_array($this->_escape_char))
1409              {
1410                  $preg_ec = array(
1411                      preg_quote($this->_escape_char[0], '/'),
1412                      preg_quote($this->_escape_char[1], '/'),
1413                      $this->_escape_char[0],
1414                      $this->_escape_char[1]
1415                  );
1416              }
1417              else
1418              {
1419                  $preg_ec[0] = $preg_ec[1] = preg_quote($this->_escape_char, '/');
1420                  $preg_ec[2] = $preg_ec[3] = $this->_escape_char;
1421              }
1422          }
1423  
1424          foreach ($this->_reserved_identifiers as $id)
1425          {
1426              if (strpos($item, '.'.$id) !== FALSE)
1427              {
1428                  return preg_replace('/'.$preg_ec[0].'?([^'.$preg_ec[1].'\.]+)'.$preg_ec[1].'?\./i', $preg_ec[2].'$1'.$preg_ec[3].'.', $item);
1429              }
1430          }
1431  
1432          return preg_replace('/'.$preg_ec[0].'?([^'.$preg_ec[1].'\.]+)'.$preg_ec[1].'?(\.)?/i', $preg_ec[2].'$1'.$preg_ec[3].'$2', $item);
1433      }

1543      /**
1544       * Returns the SQL string operator
1545       *
1546       * @param    string
1547       * @return    string
1548       */
1549  	protected function _get_operator($str)
1550      {
1551          static $_operators;
1552  
1553          if (empty($_operators))
1554          {
1555              $_les = ($this->_like_escape_str !== '')
1556                  ? '\s+'.preg_quote(trim(sprintf($this->_like_escape_str, $this->_like_escape_chr)), '/')
1557                  : '';
1558              $_operators = array(
1559                  '\s*(?:<|>|!)?=\s*',             // =, <=, >=, !=
1560                  '\s*<>?\s*',                     // <, <>
1561                  '\s*>\s*',                       // >
1562                  '\s+IS NULL',                    // IS NULL
1563                  '\s+IS NOT NULL',                // IS NOT NULL
1564                  '\s+EXISTS\s*\(.*\)',        // EXISTS(sql)
1565                  '\s+NOT EXISTS\s*\(.*\)',    // NOT EXISTS(sql)
1566                  '\s+BETWEEN\s+',                 // BETWEEN value AND value
1567                  '\s+IN\s*\(.*\)',            // IN(list)
1568                  '\s+NOT IN\s*\(.*\)',        // NOT IN (list)
1569                  '\s+LIKE\s+\S.*('.$_les.')?',    // LIKE 'expr'[ ESCAPE '%s']
1570                  '\s+NOT LIKE\s+\S.*('.$_les.')?' // NOT LIKE 'expr'[ ESCAPE '%s']
1571              );
1572  
1573          }
1574  
1575          return preg_match('/'.implode('|', $_operators).'/i', $str, $match)
1576              ? $match[0] : FALSE;
1577      }

1728       * Display an error message
1729       *
1730       * @param    string    the error message
1731       * @param    string    any "swap" values
1732       * @param    bool    whether to localize the message
1733       * @return    string    sends the application/views/errors/error_db.php template
1734       */
1735  	public function display_error($error = '', $swap = '', $native = FALSE)
1736      {
1737          $LANG =& load_class('Lang', 'core');
1738          $LANG->load('db');
1739  
1740          $heading = $LANG->line('db_error_heading');
1741  
1742          if ($native === TRUE)
1743          {
1744              $message = (array) $error;
1745          }
1746          else
1747          {
1748              $message = is_array($error) ? $error : array(str_replace('%s', $swap, $LANG->line($error)));
1749          }
1750  
1751          // Find the most likely culprit of the error by going through
1752          // the backtrace until the source file is no longer in the
1753          // database folder.
1754          $trace = debug_backtrace();
1755          foreach ($trace as $call)
1756          {
1757              if (isset($call['file'], $call['class']))
1758              {
1759                  // We'll need this on Windows, as APPPATH and BASEPATH will always use forward slashes
1760                  if (DIRECTORY_SEPARATOR !== '/')
1761                  {
1762                      $call['file'] = str_replace('\\', '/', $call['file']);
1763                  }
1764  
1765                  if (strpos($call['file'], BASEPATH.'database') === FALSE && strpos($call['class'], 'Loader') === FALSE)
1766                  {
1767                      // Found it - use a relative path for safety
1768                      $message[] = 'Filename: '.str_replace(array(APPPATH, BASEPATH), '', $call['file']);
1769                      $message[] = 'Line Number: '.$call['line'];
1770                      break;
1771                  }
1772              }
1773          }
1774  
1775          $error =& load_class('Exceptions', 'core');
1776          echo $error->show_error($heading, $message, 'error_db');
1777          exit(8); // EXIT_DATABASE
1778      }

1782      /**
1783       * Protect Identifiers
1784       *
1785       * This function is used extensively by the Query Builder class, and by
1786       * a couple functions in this class.
1787       * It takes a column or table name (optionally with an alias) and inserts
1788       * the table prefix onto it. Some logic is necessary in order to deal with
1789       * column names that include the path. Consider a query like this:
1790       *
1791       * SELECT hostname.database.table.column AS c FROM hostname.database.table
1792       *
1793       * Or a query with aliasing:
1794       *
1795       * SELECT m.member_id, m.member_name FROM members AS m
1796       *
1797       * Since the column name can include up to four segments (host, DB, table, column)
1798       * or also have an alias prefix, we need to do a bit of work to figure this out and
1799       * insert the table prefix (if it exists) in the proper position, and escape only
1800       * the correct identifiers.
1801       *
1802       * @param    string
1803       * @param    bool
1804       * @param    mixed
1805       * @param    bool
1806       * @return    string
1807       */
1808  	public function protect_identifiers($item, $prefix_single = FALSE, $protect_identifiers = NULL, $field_exists = TRUE)
1809      {
1810          if ( ! is_bool($protect_identifiers))
1811          {
1812              $protect_identifiers = $this->_protect_identifiers;
1813          }
1814  
1815          if (is_array($item))
1816          {
1817              $escaped_array = array();
1818              foreach ($item as $k => $v)
1819              {
1820                  $escaped_array[$this->protect_identifiers($k)] = $this->protect_identifiers($v, $prefix_single, $protect_identifiers, $field_exists);
1821              }
1822  
1823              return $escaped_array;
1824          }
1825  
1826          // This is basically a bug fix for queries that use MAX, MIN, etc.
1827          // If a parenthesis is found we know that we do not need to
1828          // escape the data or add a prefix. There's probably a more graceful
1829          // way to deal with this, but I'm not thinking of it -- Rick
1830          //
1831          // Added exception for single quotes as well, we don't want to alter
1832          // literal strings. -- Narf
1833          if (strcspn($item, "()'") !== strlen($item))
1834          {
1835              return $item;
1836          }
1837  
1838          // Convert tabs or multiple spaces into single spaces
1839          $item = preg_replace('/\s+/', ' ', trim($item));
1840  
1841          // If the item has an alias declaration we remove it and set it aside.
1842          // Note: strripos() is used in order to support spaces in table names
1843          if ($offset = strripos($item, ' AS '))
1844          {
1845              $alias = ($protect_identifiers)
1846                  ? substr($item, $offset, 4).$this->escape_identifiers(substr($item, $offset + 4))
1847                  : substr($item, $offset);
1848              $item = substr($item, 0, $offset);
1849          }
1850          elseif ($offset = strrpos($item, ' '))
1851          {
1852              $alias = ($protect_identifiers)
1853                  ? ' '.$this->escape_identifiers(substr($item, $offset + 1))
1854                  : substr($item, $offset);
1855              $item = substr($item, 0, $offset);
1856          }
1857          else
1858          {
1859              $alias = '';
1860          }
1861  
1862          // Break the string apart if it contains periods, then insert the table prefix
1863          // in the correct location, assuming the period doesn't indicate that we're dealing
1864          // with an alias. While we're at it, we will escape the components
1865          if (strpos($item, '.') !== FALSE)
1866          {
1867              $parts = explode('.', $item);
1868  
1869              // Does the first segment of the exploded item match
1870              // one of the aliases previously identified? If so,
1871              // we have nothing more to do other than escape the item
1872              //
1873              // NOTE: The ! empty() condition prevents this method
1874              //       from breaking when QB isn't enabled.
1875              if ( ! empty($this->qb_aliased_tables) && in_array($parts[0], $this->qb_aliased_tables))
1876              {
1877                  if ($protect_identifiers === TRUE)
1878                  {
1879                      foreach ($parts as $key => $val)
1880                      {
1881                          if ( ! in_array($val, $this->_reserved_identifiers))
1882                          {
1883                              $parts[$key] = $this->escape_identifiers($val);
1884                          }
1885                      }
1886  
1887                      $item = implode('.', $parts);
1888                  }
1889  
1890                  return $item.$alias;
1891              }
1892  
1893              // Is there a table prefix defined in the config file? If not, no need to do anything
1894              if ($this->dbprefix !== '')
1895              {
1896                  // We now add the table prefix based on some logic.
1897                  // Do we have 4 segments (hostname.database.table.column)?
1898                  // If so, we add the table prefix to the column name in the 3rd segment.
1899                  if (isset($parts[3]))
1900                  {
1901                      $i = 2;
1902                  }
1903                  // Do we have 3 segments (database.table.column)?
1904                  // If so, we add the table prefix to the column name in 2nd position
1905                  elseif (isset($parts[2]))
1906                  {
1907                      $i = 1;
1908                  }
1909                  // Do we have 2 segments (table.column)?
1910                  // If so, we add the table prefix to the column name in 1st segment
1911                  else
1912                  {
1913                      $i = 0;
1914                  }
1915  
1916                  // This flag is set when the supplied $item does not contain a field name.
1917                  // This can happen when this function is being called from a JOIN.
1918                  if ($field_exists === FALSE)
1919                  {
1920                      $i++;
1921                  }
1922  
1923                  // dbprefix may've already been applied, with or without the identifier escaped
1924                  $ec = '(?<ec>'.preg_quote(is_array($this->_escape_char) ? $this->_escape_char[0] : $this->_escape_char).')?';
1925                  isset($ec[0]) && $ec .= '?'; // Just in case someone has disabled escaping by forcing an empty escape character
1926  
1927                  // Verify table prefix and replace if necessary
1928                  if ($this->swap_pre !== '' && preg_match('#^'.$ec.preg_quote($this->swap_pre).'#', $parts[$i]))
1929                  {
1930                      $parts[$i] = preg_replace('#^'.$ec.preg_quote($this->swap_pre).'(\S+?)#', '\\1'.$this->dbprefix.'\\2', $parts[$i]);
1931                  }
1932                  // We only add the table prefix if it does not already exist
1933                  else
1934                  {
1935                      preg_match('#^'.$ec.preg_quote($this->dbprefix).'#', $parts[$i]) OR $parts[$i] = $this->dbprefix.$parts[$i];
1936                  }
1937  
1938                  // Put the parts back together
1939                  $item = implode('.', $parts);
1940              }
1941  
1942              if ($protect_identifiers === TRUE)
1943              {
1944                  $item = $this->escape_identifiers($item);
1945              }
1946  
1947              return $item.$alias;
1948          }
1949  
1950          // Is there a table prefix? If not, no need to insert it
1951          if ($this->dbprefix !== '')
1952          {
1953              // Verify table prefix and replace if necessary
1954              if ($this->swap_pre !== '' && strpos($item, $this->swap_pre) === 0)
1955              {
1956                  $item = preg_replace('/^'.$this->swap_pre.'(\S+?)/', $this->dbprefix.'\\1', $item);
1957              }
1958              // Do we prefix an item with no segments?
1959              elseif ($prefix_single === TRUE && strpos($item, $this->dbprefix) !== 0)
1960              {
1961                  $item = $this->dbprefix.$item;
1962              }
1963          }
1964  
1965          if ($protect_identifiers === TRUE && ! in_array($item, $this->_reserved_identifiers))
1966          {
1967              $item = $this->escape_identifiers($item);
1968          }
1969  
1970          return $item.$alias;
1971      }
