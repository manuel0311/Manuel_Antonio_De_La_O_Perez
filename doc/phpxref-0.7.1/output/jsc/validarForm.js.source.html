 179  function validarDNI() {
 180  
 181      dni = document.getElementById("DNI").value.toUpperCase();
 182      expresionregulardni=new RegExp(/^[0-9]{8}[A-Z]{1}$/);
 183  
 184      if(expresionregulardni.test(dni))
 185      {
 186  
 187          //Se separan los números de la letra
 188          var letraDNI = dni.substring(8, 9).toUpperCase();
 189          var numDNI = parseInt(dni.substring(0, 8));
 190  
 191          //Se calcula la letra correspondiente al número
 192          var letras = ['T', 'R', 'W', 'A', 'G', 'M', 'Y', 'F', 'P', 'D', 'X', 'B', 'N', 'J', 'Z', 'S', 'Q', 'V', 'H', 'L', 'C', 'K', 'E', 'T'];
 193          var letraCorrecta = letras[numDNI % 23];
 194  
 195          if (letraDNI != letraCorrecta) {
 196              document.getElementById("DNI").style.backgroundColor="ffb0b0";
 197              DNIValido=false;
 198              document.getElementById("DNI").focus();
 199              document.getElementById("errorDNI").innerHTML =  "Letra incorrecta,por favor,inténtelo de nuevo";

 201              compruebaDNI(dni);
