  60  function setCookie(cname, cvalue, exdays) {
  61      // expire the old cookie if existed to avoid multiple cookies with the same name
  62      if  (getCookie(cname)) {
  63          document.cookie = cname + "=;expires=Thu, 01 Jan 1970 00:00:00 GMT";
  64      }
  65      var d = new Date();
  66      d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
  67      var expires = "expires=" + d.toGMTString();
  68      document.cookie = cname + "=" + cvalue + "; " + expires + "; path=/";
  69  }
  70  function getCookie(cname) {
  71      var name = cname + "=";
  72      var ca = document.cookie.split(';');
  73      for (var i = 0; i < ca.length; i++) {
  74          var c = ca[i];
  75          while (c.charAt(0) == ' ')
  76              c = c.substring(1);
  77          if (c.indexOf(name) == 0) {
  78              return c.substring(name.length, c.length);
  79          }
  80      }
  81      return '';
  82  }
